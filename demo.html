show success | show error | show warning | show info | show debug | show using options | show code 
      function showSuccess() {
        $.n("This is a success message");
      }

      function showError() {
        $.n.error("This is an error message. It won't go away you have to click the x button.");
      }

      function showWarning() {
        $.n.warning("This is a warning message");
      }

      function showInfo() {
        $.n.info("This is an info message. It will automatically go away.");
      }

      function showDebug() {
        $.n.info("This is a debug message. It won't go away you have to click the x button.");
      }

      function showWithOptions() {
        $.n("This message has options.", {
          type: "success",
          timeout: 5000,
          stick: true,
          fadeSpeed : 1500,
          close : "close me",
          effect : "slide"})
      }
